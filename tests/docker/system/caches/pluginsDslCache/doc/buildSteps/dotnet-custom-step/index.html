A dotnet build step to run a custom command

Example. Runs dotnet EXECUTABLE to run specified .NET application

buildType {
    // Other Build Type settings ...
    steps {
        // Other Build Steps ...
        dotnetCustom {
            executables = "MyProgram.exe"
            args = "abc=123"
        }
